```python
# 错误方法（找顶层 role）：
role = d.get('role','')  # ❌ 永远是空

# 正确方法（message 嵌套结构）：
if d.get('type') == 'message':
    msg = d.get('message', {})
    role = msg.get('role','')  # ✅ user/assistant/toolResult
    content = msg.get('content','')  # list of {type: 'text'/'thinking'/'toolCall'}
```
